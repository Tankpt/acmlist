#include<stdio.h>
#include<math.h>
int main() 
{
	int n,sum,i,k,j,num[1000];
	while(scanf("%d",&n)!=EOF && n!=0)
	{
		for(i=0;i<n;i++)
			scanf("%d",&num[i]);
		for(i=0;i<n;i++)
		{
			k=num[i];sum=0;
			for(j=2;j<=k;j++)
			{
				if(k%j==0) sum++;
			}
			printf("%d\n",sum+1);
		}
	}
	return 0;
}

#include<stdio.h>
#include<math.h>
int main() 
{
	int n,sum,i,k,j,num[1000];
	while(scanf("%d",&n)!=EOF && n!=0)
	{
		for(i=0;i<n;i++)
			scanf("%d",&num[i]);
		for(i=0;i<n;i++)
		{
			k=sqrt(num[i]);sum=0;
			for(j=1;j<=k;j++)
			{
				if(num[i]%j==0) sum++;
			}
			if(k*k==num[i]) 
				printf("%d\n",2*sum-1);
			else 
				printf("%d\n",2*sum);;
		}
	}
	return 0;
}
