#include <stdio.h>
int tree[1001];
int find(int x)
{
	int t,ret,tmp;
	tmp=x;
	while(tree[x]!=-1)
		x=tree[x];
	ret=x;x=tmp;
	while(tree[x]!=-1)
	{
		t=tree[x];
		tree[x]=ret;
		x=t;
	}
	return ret;
};
int main() 
{
	int t,i,n,m,a,b,j,sum;
	char c;
	while( scanf("%d",&t)!=EOF)
	{
		for(j=0;j<t;j++)
		{
			scanf("%d%d",&n,&m);
			for(i=1;i<=n;i++)
				tree[i]=-1;
			for(i=0;i<m;i++)
			{
				scanf("%d%d",&a,&b);
				a=find(a);b=find(b);
				if(a!=b)
					tree[a]=b;
			}
			sum=0;
			for(i=1;i<=n;i++)
				if(tree[i]==-1) sum++;
			printf("%d\n",sum);
			if(j!=t-1) c=getchar();
		}

	}
	return 0;

}
