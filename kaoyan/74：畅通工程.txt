样例输入：
3 3
1 2 1
1 3 2
2 3 4
1 3
2 3 2
0 100
样例输出：
3
?


#include <stdio.h>
#include <algorithm>
#define N 101
using namespace std;
int tree[N];
struct Edge{
	int a;int b;
	int cost;
}edge[N];
bool cmp(Edge a,Edge b)
{
	return a.cost<b.cost;
};
int find(int x)
{
	int t,ret,tmp;
	tmp=x;
	while(tree[x]>0)
		x=tree[x];
	ret=x;x=tmp;
	while(tree[x]>0)
	{
		t=tree[x];
		tree[x]=ret;
		x=t;
	}
	return ret;
};
int main() 
{
	int n,m,i,j;
	int a,b,sum;
	while( scanf("%d%d",&n,&m)!=EOF && n!=0 )
	{
		for(i=1;i<=m;i++)
			tree[i]=-1;
		for(i=0;i<n;i++)
			scanf("%d%d%d",&edge[i].a,&edge[i].b,&edge[i].cost);
		sort(edge,edge+n,cmp);
		sum=0;
		for(i=0;i<n;i++)
		{
			a=edge[i].a;b=edge[i].b;
			a=find(a);b=find(b);
			if(a!=b)
			{
				tree[a]=b;
				sum=sum+edge[i].cost;
			}
		}
		a=0;
		for(i=1;i<=m;i++)
			if(tree[i]==-1) a++;
		if(a==1)
			printf("%d\n",sum);
		else printf("?\n");
	}
	return 0;
}
