#include<stdio.h>
#include<stdlib.h>
#include<string.h>
int main() 
{
	long long  n,j,sum;
	while(scanf("%lld",&n)!=EOF && n!=0)
	{
		while(n/10!=0)
		{
			j=n;sum=0;
			while(j>0)
			{
				sum=sum+j%10;
				j=j/10;
			}
			n=sum;
		}
		printf("%lld\n",n);
	}

	return 0;
}

#include<stdio.h>
#include<stdlib.h>
#include<string.h>
int main() 
{
	int i,sum,n;
	char m[40];
	while(scanf("%s",&m)!=EOF )
	{
		sum=0;
		for(i=0;m[i]!='\0';i++)
			sum=sum+m[i]-'0';
		n=sum;
		if(n==0) break;
		while(n/10!=0)
		{
			i=n;sum=0;
			while(i>0)
			{
				sum=sum+i%10;
				i=i/10;
			}
			n=sum;
		}
		printf("%d\n",n);
	}

	return 0;
}
