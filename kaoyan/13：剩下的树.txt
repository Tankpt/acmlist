#include <stdio.h>
#include <algorithm>

using namespace std;
int L[10002];
int main()
{
	int l,m,i,beg,end,count,j;
	while(scanf("%d%d",&l,&m)!=EOF)
	{
		int L[10002]={0};//为啥赋值1不行
		for(i=0;i<m;i++)
		{
			scanf("%d%d",&beg,&end);
			for(j=beg;j<=end;j++)
				L[j]=1;
		}
		count=0;
		for(i=0;i<=l;i++)
			if(L[i]==0) count++;
		printf("%d\n",count);
	}
	return 0;
}
